Here is the JavaScript ode I wrote to implement this algorithm.

        
 
            
                var deleteDuplicates = function(head) {
                    if (head == null || head.next == null) return head;
                    var prev = head, curr = head.next, after = curr.next;
                    while(curr != null){
                        if(curr.val == prev.val){
                            prev.next = curr.next;
                            curr = curr.next;
                        }
                        else {
                            var temp = curr;
                            curr = curr.next;
                            prev = temp;
                        }
                    }
                    return head;
                }
